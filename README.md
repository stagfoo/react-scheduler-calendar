## Install

```bash
npm install --save react-scheduler-calendar
```

## Usage

```tsx
import React from 'react'

import Scheduler from 'react-scheduler-calendar'

class Example extends React.Component {
  render () {
    return (
      <Scheduler />
    )
  }
}
```
